# [1b] NeRF-synthetic 데이터셋 생성 — RGBA(알파=th=12 마스크) + 원형 transforms
# 로더 계약(third_party/Relightable3DGaussian/scene/dataset_readers.py 검증):
#  - 마스크 = RGBA 알파채널 (별도 masks/ 아님)  - file_path는 확장자 없이(로더가 .png append)
#  - transform_matrix = OpenGL/Blender c2w (로더가 c2w[:,1:3]*=-1로 COLMAP 변환)  - 배경 검정 기본
# canonical 헬퍼: src/poses_turntable.py (여기선 self-contained 인라인으로 기록)
import os, json, math, glob
import numpy as np
from PIL import Image

base='data/peter_body'; img_dir=base+'/images'; rgba_dir=base+'/rgba'
os.makedirs(rgba_dir, exist_ok=True)
TH=12; radius, elevation_deg, fov_x_deg, test_every = 3.0, 15.0, 40.0, 8

# (1) RGBA: alpha = luminance mask
files=sorted(f for f in os.listdir(img_dir) if f.endswith('.png'))
fracs=[]
for fn in files:
    rgb=np.array(Image.open(os.path.join(img_dir,fn)).convert('RGB'))
    alpha=((rgb.mean(-1)>TH)*255).astype('uint8')
    Image.fromarray(np.dstack([rgb,alpha]),'RGBA').save(os.path.join(rgba_dir,fn))
    fracs.append((alpha>0).mean())

# (2) 원형 poses — 절대 인덱스로 theta, every-8th held-out
def look_at_c2w(eye, target=(0,0,0), up=(0,0,1)):
    eye,target,up=map(lambda v: np.array(v,float),(eye,target,up))
    f=target-eye; f/=np.linalg.norm(f)
    r=np.cross(f,up); r/=np.linalg.norm(r); u=np.cross(r,f)
    c2w=np.eye(4); c2w[:3,0],c2w[:3,1],c2w[:3,2],c2w[:3,3]=r,u,-f,eye
    return c2w
n=len(files); el=math.radians(elevation_deg); train,test=[],[]
for i,fn in enumerate(files):
    th=2*math.pi*i/n
    eye=(radius*math.cos(th)*math.cos(el), radius*math.sin(th)*math.cos(el), radius*math.sin(el))
    fr={'file_path': os.path.join('.','rgba',os.path.splitext(fn)[0]), 'transform_matrix': look_at_c2w(eye).tolist()}
    (test if i%test_every==0 else train).append(fr)
for name,frames in (('transforms_train.json',train),('transforms_test.json',test)):
    json.dump({'camera_angle_x':math.radians(fov_x_deg),'frames':frames}, open(os.path.join(base,name),'w'), indent=2)

print('RGBA',len(files),'| fg frac %.3f (%.3f-%.3f)'%(np.mean(fracs),min(fracs),max(fracs)))
print('transforms: train',len(train),'test',len(test))
print('loader path check:', os.path.exists(os.path.join(base, train[0]['file_path']+'.png')))
#+end_src
